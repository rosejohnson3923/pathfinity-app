import { useStudentProfile } from './hooks/useStudentProfile';

function StudentDashboard({ userId, userEmail }) {
  const {
    profile,
    loading,
    displayName,
    updateGradeLevel
  } = useStudentProfile(userId, userEmail, true); // Auto-create enabled

          <button onClick={() => updateGradeLevel('K')}>

          </button>
        </div>
      )}
    </div>
  );
}
